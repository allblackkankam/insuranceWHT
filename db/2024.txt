
CREATE TABLE `insurancedb`.`insurance` (`id` INT NOT NULL AUTO_INCREMENT , `facility_id` VARCHAR(200) NOT NULL DEFAULT '' , `insurance_name` VARCHAR(200) NOT NULL DEFAULT '' , `insurance_code` VARCHAR(50) NOT NULL DEFAULT '' , `insurance_status` TINYINT(1) NOT NULL DEFAULT '0' , `date` DATETIME NOT NULL , PRIMARY KEY (`id`)) ENGINE = MyISAM;
